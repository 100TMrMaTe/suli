dashboard.blade.php létrehozása resources > views
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<body>
    <h1>API test</h1>
    <button onclick="load()">meghiv</button>
    <p id="eredmeny"></p>
    <script>
        function load() {
            fetch("/api/test")
                .then(response => response.json())
                .then(data => {
                    console.log(data);
                    document.getElementById("eredmeny").innerText = data.message;
                });
        }
    </script>
</body>
</html>
